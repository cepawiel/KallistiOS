11 /** \file dc/maple.h
12  \brief Maple Bus driver interface.
13  \ingroup maple
14 
15  This file provides support for accessing the Maple bus on the Dreamcast.
16  Maple is the bus that all of your controllers and memory cards and the like
17  connect to, so this is one of those types of things that are quite important
18  to know how to use.
19

21  accessed through the functions here. Most of the drivers have their own
22  functionality that is implemented in their header files, as well.

863 /** \brief Begin a foreach loop over Maple devices.
864  \ingroup maple
865 
866  This macro (along with the MAPLE_FOREACH_END() one) implements a simple
867  foreach-style loop over the given type of devices. Essentially, it grabs the
868  status of the device, and leaves it to you to figure out what to do with it.
869 
870  The most common use of this would be to look for input on any controller.
871 
872  \param TYPE The function code of devices to look at.
873  \param VARTYPE The type to cast the return value of
874  maple_dev_status() to.
875  \param VAR The name of the result of maple_dev_status().
876 */
877 #define MAPLE_FOREACH_BEGIN(TYPE, VARTYPE, VAR) \
878  do { \
879  maple_device_t * __dev; \
880  VARTYPE * VAR; \
881  int __i; \
882  \
883  __i = 0; \
884  while( (__dev = maple_enum_type(__i, TYPE)) ) { \
885  VAR = (VARTYPE *)maple_dev_status(__dev); \
886  do {

889  \ingroup maple
890 
891  Each MAPLE_FOREACH_BEGIN() must be paired with one of these after the loop
892  body.
893 */
894 #define MAPLE_FOREACH_END() \
895  } while(0); \

897  } \
898  } while(0);
