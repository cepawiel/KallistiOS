10 /** \file dc/sound/sfxmgr.h
11  \brief Basic sound effect support.
12  \ingroup audio_sfx
13 
14  This file contains declarations for doing simple sound effects. This code is
15  only usable for simple WAV files containing either 8-bit or 16-bit samples (stereo
16  or mono) or Yamaha ADPCM (4-bits, stereo or mono). Also, all sounds played in
17  this manner must be at most 65534 samples in length, as this does not handle
18  buffer chaining or anything else complex. For more interesting stuff, you
19  should probably look at the sound stream stuff instead.
20 
21  \author Megan Potter
22  \author Ruslan Rostovtsev
23  \author Andy Barajas
24 */

42 /** \brief Sound effect handle type.
43 
44  Each loaded sound effect will be assigned one of these, which is to be used
45  for operations related to the effect, including playing it or unloading it.
46 */
47 typedef uint32_t sfxhnd_t;

49 /** \brief Invalid sound effect handle value.
50 
51  If a sound effect cannot be loaded, this value will be returned as the error
52  condition.
53 */
54 #define SFXHND_INVALID 0

56 /** \brief Load a sound effect.
57 
58  This function loads a sound effect from a WAV file and returns a handle to
59  it. The sound effect can be either stereo or mono, and must either be 8-bit
60  or 16-bit uncompressed PCM samples, or 4-bit Yamaha ADPCM.
61 
62  \warning The sound effect you are loading must be at most 65534 samples
63  in length.
64 
65  \param fn The file to load.
66  \return A handle to the sound effect on success. On error,
67  SFXHND_INVALID is returned.
68 */
69 sfxhnd_t snd_sfx_load(const char *fn);
70 
71 /** \brief Unload a sound effect.
72 
73  This function unloads a previously loaded sound effect, and frees the memory
74  associated with it.
75 
76  \param idx A handle to the sound effect to unload.
77 */
78 void snd_sfx_unload(sfxhnd_t idx);
79 
80 /** \brief Unload all loaded sound effects.
81 
82  This function unloads all previously loaded sound effect, and frees the
83  memory associated with them.
84 */
85 void snd_sfx_unload_all(void);

87 /** \brief Play a sound effect.
88 
89  This function plays a loaded sound effect with the specified volume (for
90  both stereo or mono) and panning values (for mono sounds only).
91 
92  \param idx The handle to the sound effect to play.
93  \param vol The volume to play at (between 0 and 255).
94  \param pan The panning value of the sound effect. 0 is all the
95  way to the left, 128 is center, 255 is all the way
96  to the right.
97 
98  \return The channel used to play the sound effect (or the
99  left channel in the case of a stereo sound, the
100  right channel will be the next one) on success, or
101  -1 on failure.
102 */
103 int snd_sfx_play(sfxhnd_t idx, int vol, int pan);
104 
105 /** \brief Play a sound effect on a specific channel.
106 
107  This function works similar to snd_sfx_play(), but allows you to specify the
108  channel to play on. No error checking is done with regard to the channel, so
109  be sure its safe to play on that channel before trying.
110 
111  \param chn The channel to play on (or in the case of stereo,
112  the left channel).
113  \param idx The handle to the sound effect to play.
114  \param vol The volume to play at (between 0 and 255).
115  \param pan The panning value of the sound effect. 0 is all the
116  way to the left, 128 is center, 255 is all the way
117  to the right.
118 
119  \return chn
120 */
121 int snd_sfx_play_chn(int chn, sfxhnd_t idx, int vol, int pan);
122 
123 /** \brief Stop a single channel of sound.
124 
125  This function stops the specified channel of sound from playing. It does no
126  checking to make sure that a sound effect is playing on the channel
127  specified, and thus can be used even if you're using the channel for some
128  other purpose than sound effects.
129 
130  \param chn The channel to stop.
131 */
132 void snd_sfx_stop(int chn);
133 
134 /** \brief Stop all channels playing sound effects.
135 
136  This function stops all channels currently allocated to sound effects from
137  playing. It does not affect channels allocated for use by something other
138  than sound effects..
139 */
140 void snd_sfx_stop_all(void);
141 
142 /** \brief Allocate a sound channel for use outside the sound effect system.
143 
144  This function finds and allocates a channel for use for things other than
145  sound effects. This is useful for, for instance, the streaming code.
146 
147  \returns The allocated channel on success, -1 on failure.
148 */
149 int snd_sfx_chn_alloc(void);
150 
151 /** \brief Free a previously allocated channel.
152 
153  This function frees a channel that was allocated with snd_sfx_chn_alloc(),
154  returning it to the pool of available channels for sound effect use.
155 
156  \param chn The channel to free.
157 */
158 void snd_sfx_chn_free(int chn);
